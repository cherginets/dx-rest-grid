## Install

```bash
npm install --save dx-rest-grid
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'dx-rest-grid'
import 'dx-rest-grid/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
